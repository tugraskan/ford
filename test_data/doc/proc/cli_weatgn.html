DO n = 1, 3 
wgncur(n,iwgn) = 0.
DO l = 1, 3 
wgncur(n,iwgn) = wgncur(n,iwgn) + b(n,l) * e(l)
xx(n) = xx(n) + a(n,l) * wgnold(l,iwgn)
DO n = 1, 3 
wgncur(n,iwgn) = wgncur(n,iwgn) + xx(n)
if (wgncur(n,iwgn) > 1.) wgncur(n,iwgn) = 1.
if (wgncur(n,iwgn) < -1.) wgncur(n,iwgn) = -1.
wgnold(n,iwgn) = wgncur(n,iwgn)
